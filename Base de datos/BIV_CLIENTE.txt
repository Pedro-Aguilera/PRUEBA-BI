CREATE OR REPLACE VIEW BIV_CLIENTE AS
SELECT C.CODIGO_CLIENTE,
       C.NOMBRE_CLIENTE,
       C.TIPO_PERSONA,
       C.COD_GENERO GENERO,
       C.FECHA_NACIMIENTO,
       C.LOCALIDAD,
       C.FECHA_ALTA_CLIENTE,
       G.DESCRIPCION DESC_GENERO,
       ROUND((SYSDATE - C.FECHA_NACIMIENTO)/365) EDAD
FROM CLIENTES C,
     GENERO   G
WHERE C.COD_GENERO = G.COD_GENERO;
