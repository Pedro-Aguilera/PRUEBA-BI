CREATE OR REPLACE VIEW BIV_SALDO_PRODUCTOS AS
SELECT P.OPERACION,
       P.CODIGO_CLIENTE,
       P.COD_MONEDA,
       P.TIPO_CARTERA,
       P.SALDO_CAPITAL,
       P.SALDO_INTERES,
       P.FECHA_PROCESO,
       P.FECHA_NACIMIENTO,
       (P.SALDO_CAPITAL + P.SALDO_INTERES) SALDO_PRODUCTO,
       M.DESCRIPCION DESC_MONEDA,
       M.COTIZACION
FROM PRODUCTOS P,
     MONEDA    M
WHERE P.COD_MONEDA = M.COD_MONEDA;
